# Build is too large, running in dev mode instead

cd ~/apps/darkbyte-dashboard
git pull origin main
npm install
npx prisma generate
npx prisma migrate deploy

# Run in development mode (faster, no build needed)
pm2 delete all
pm2 start "npm run dev" --name "dashboard"
pm2 save
pm2 logs dashboard --lines 20

=== FIRST TIME VPS SETUP ===
# Install dependencies
sudo apt update && sudo apt upgrade -y
curl -fsSL https://deb.nodesource.com/setup_20.x | sudo -E bash -
sudo apt install -y nodejs postgresql nginx
sudo npm install -g pm2

# PostgreSQL setup
sudo -u postgres psql -c "CREATE DATABASE darkbyte;"
sudo -u postgres psql -c "CREATE USER dbuser WITH PASSWORD 'password123';"
sudo -u postgres psql -c "GRANT ALL PRIVILEGES ON DATABASE darkbyte TO dbuser;"

# Clone repo
cd /var/www
git clone YOUR_GITHUB_REPO app
cd app
cp .env.example .env
nano .env  # Edit with production values
npm install
npx prisma generate
npx prisma migrate deploy
npm run build
pm2 start npm --name "dashboard" -- start
pm2 startup
pm2 save
